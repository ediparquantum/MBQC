24std::vector<std::vector<int>> create_lattice(int numRows, int numCols) {
25 int numVertices = numRows * numCols;
26 std::vector<std::vector<int>> lattice(numRows, std::vector<int>(numCols, -1));
27 std::vector<int> numbers(numVertices);
28 for (int i = 0; i < numVertices; i++) {
29 numbers[i] = i;
30 }
31
32 for (int number : numbers) {
33 int row = number % numRows;
34 int col = number / numRows;
35 lattice[row][col] = number;
36 }
37
38 return lattice;
39}

54undirectedGraph createLatticeGraph(int numRows, int numCols) {
55 std::vector<std::vector<int>> lattice = create_lattice(numRows, numCols);
56 int numVertices = numRows * numCols;
57 undirectedGraph latticeGraph(numVertices);
58
59 // Add edges to the graph
60 for (int row = 0; row < numRows; row++) {
61 for (int col = 0; col < numCols; col++) {
62 if (row + 1 < numRows) {
63 boost::add_edge(lattice[row][col], lattice[row + 1][col], latticeGraph);
64 }
65 if (col + 1 < numCols) {
66 boost::add_edge(lattice[row][col], lattice[row][col + 1], latticeGraph);
67 }
68 }
69 }
70
71 return latticeGraph;
72}

180undirectedGraph createCompleteMultipartiteGraph(const std::vector<int>& partitions) {
181 int numPartitions = partitions.size();
182 int numVertices = 0;
183 for (int partitionSize : partitions) {
184 numVertices += partitionSize;
185 }
186
187 undirectedGraph graph(numVertices);
188
189 int vertexOffset = 0;
190 for (int partitionSize : partitions) {
191 for (int i = vertexOffset; i < vertexOffset + partitionSize; ++i) {
192 for (int j = vertexOffset + partitionSize; j < numVertices; ++j) {
193 boost::add_edge(i, j, graph);
194 }
195 }
196 vertexOffset += partitionSize;
197 }
198
199 return graph;
200}
