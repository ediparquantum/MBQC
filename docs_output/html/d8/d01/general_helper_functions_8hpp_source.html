91std::vector<double> calculateColumnRatios(const std::vector<std::vector<int>>& matrix) {
92 std::vector<double> ratios;
93
94 if (matrix.empty() || matrix[0].empty()) {
95 return ratios; // Empty matrix, return empty ratios vector
96 }
97
98 const int numRows = matrix.size();
99 const int numCols = matrix[0].size();
100
101 for (int col = 0; col < numCols; col++) {
102 int count = 0;
103 for (int row = 0; row < numRows; row++) {
104 if (matrix[row][col] == 1) {
105 count++;
106 }
107 }
108 double ratio = static_cast<double>(count) / numRows;
109 ratios.push_back(ratio);
110 }
111
112 return ratios;
113}

131std::vector<std::string> getPaths(std::string dataPath, std::string outputDir, std::string qubitInfo, qreal probability) {
132 const char* homeDir = std::getenv("HOME");
133 std::string fslash = "/";
134 std::string uscore = "_";
135 std::string root = homeDir;
136 std::string amps = "amplitudes";
137 std::string probs = "probabilities";
138 std::string damps;
139 std::string csv = ".csv";
140
141
142 if(probability < 0){
143 damps = "";
144 } else{
145 int precisionVal = 2;
146 std::string prob_str = std::to_string(probability).substr(0, std::to_string(probability).find(".") + precisionVal + 1);
147 damps = "mix_damping_prob-"+prob_str;
148 }
149
150 std::string dataDirPath = homeDir + fslash + dataPath + fslash + outputDir + fslash;
151 std::string fileAmp = dataDirPath + amps + uscore + qubitInfo + uscore + damps + csv;
152 std::string fileProb = dataDirPath + probs + uscore + qubitInfo + uscore + damps + csv;
153
154 std::vector<std::string> paths = {fileAmp, fileProb};
155
156 return paths;
157}

278void writeStateToFile(const Qureg& qureg, const std::string& filepath) {
279 FILE* state = std::fopen(filepath.c_str(), "w");
280 long long int index;
281 if (state == nullptr) {
282 std::perror("Failed to open file");
283 return;
284 }
285
286 if (qureg.chunkId == 0) {
287 std::fprintf(state, "real, imag\n");
288 }
289
290 for(index=0; index<qureg.numAmpsPerChunk; index++){
291 # if QuEST_PREC==1 || QuEST_PREC==2
292 fprintf(state, "%.12f, %.12f\n", qureg.stateVec.real[index], qureg.stateVec.imag[index]);
293 # elif QuEST_PREC == 4
294 fprintf(state, "%.12Lf, %.12Lf\n", qureg.stateVec.real[index], qureg.stateVec.imag[index]);
295 #endif
296 }
297 fclose(state);
298 std::cout << "Values written to file successfully." << std::endl;
299}
