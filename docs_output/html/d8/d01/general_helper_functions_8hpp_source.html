91std::vector<double> calculateColumnRatios(const std::vector<std::vector<int>>& matrix) {
92 std::vector<double> ratios;
93
94 if (matrix.empty() || matrix[0].empty()) {
95 return ratios; // Empty matrix, return empty ratios vector
96 }
97
98 const int numRows = matrix.size();
99 const int numCols = matrix[0].size();
100
101 for (int col = 0; col < numCols; col++) {
102 int count = 0;
103 for (int row = 0; row < numRows; row++) {
104 if (matrix[row][col] == 1) {
105 count++;
106 }
107 }
108 double ratio = static_cast<double>(count) / numRows;
109 ratios.push_back(ratio);
110 }
111
112 return ratios;
113}
