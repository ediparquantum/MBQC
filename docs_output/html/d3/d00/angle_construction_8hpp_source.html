89std::vector<qreal> initRandomPiOnNAnglesMod2kPi(int numQubits, int piDenominator) {
90 // Create a uniform distribution for the range
91 int min = 0;
92 int max = 2 * piDenominator - 1;
93 int totalPositions = 2 * piDenominator;
94
95 // Populate angle vector
96 std::vector<qreal> qubitAngles(numQubits);
97 std::vector<int> numeratorVector(numQubits);
98 for (int i = 0; i < numQubits - 1; i++) {
99 double numer = 1.0 * drawRandomUniformInteger(min, max);
100 numeratorVector[i] = numer;
101 qreal angle = (double)(numer / piDenominator) * M_PI;
102 qubitAngles[i] = angle;
103 }
104
105 // Compute sum of random numerators
106 int totalNumerator = std::accumulate(numeratorVector.begin(), numeratorVector.end(), 0);
107
108 // Calculate remaining positions
109 int remainingPositions = totalPositions - (totalNumerator % totalPositions);
110 int totalNumeratorAndRemainingModPositions = (remainingPositions + totalNumerator) % totalPositions;
111
112 assert((totalNumeratorAndRemainingModPositions == 0) && "Total numerator values should be mod total positions equal to 0, indicating a 2kPi multiple");
113
114 qreal finalAngle = (1.0) * (remainingPositions / piDenominator) * M_PI;
115 qubitAngles[numQubits - 1] = finalAngle;
116
117 return qubitAngles;
118}
