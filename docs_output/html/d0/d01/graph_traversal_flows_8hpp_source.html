77std::vector<int> getVertexNeighbours(const undirectedGraph& graph,int vertex){
78 // Choose a vertex to find its adjacent vertices
79 undirectedGraph::vertex_descriptor v = vertex;
80 // Find the range of adjacent vertices
81 typedef boost::graph_traits<undirectedGraph>::adjacency_iterator adjacency_iterator;
82 std::pair<adjacency_iterator, adjacency_iterator> neighbors = adjacent_vertices(v, graph);
83
84 std::vector<int> neighbourVector;
85
86 for (adjacency_iterator it = neighbors.first; it != neighbors.second; ++it) {
87 neighbourVector.push_back(*it);
88 }
89 return neighbourVector;

153qreal computeCorrectedAngle(
154 const undirectedGraph& graph,
155 std::vector<int> measuredOutcomes,
156 int vertex,
157 int numRows,
158 qreal angle){
159
160 std::vector<int> neighbourhood = getVertexNeighbours(graph,vertex);
161 int xCorrectionVertex;
162 std::vector<int> zCorrectionVertices;
163 qreal updatedAngle;
164
165 if (vertex < numRows){ // First column - no x correction no z correction
166 updatedAngle = angle;
167 } else if (vertex >= numRows){ // ensure we are in second column
168 xCorrectionVertex = getLattice2DPastVertexGraph(graph,vertex,numRows);
169 for (int i: neighbourhood){
170 int past;
171 if(i >= numRows){
172 past = getLattice2DPastVertexGraph(graph,i,numRows);
173 } else{
174 continue;
175 }
176 zCorrectionVertices.push_back(past);
177 }
178
179 // Remove vertex in neigbourhood if it is the vertrex
180 // we want to measure
181 zCorrectionVertices.erase(
182 std::remove(
183 zCorrectionVertices.begin(),
184 zCorrectionVertices.end(), vertex),
185 zCorrectionVertices.end());
186
187 int exponent = measuredOutcomes[xCorrectionVertex];
188
189 //std::cout << "f-1(j): " << xCorrectionVertex << std::endl;
190 //std::cout << "s_f-1(j): " << exponent << std::endl;
191 qreal xAngle = std::pow((-1), exponent)*angle;
192
193 std::vector<int> outcomesForZ;
194 for (int i:zCorrectionVertices){
195 outcomesForZ.push_back(measuredOutcomes[i]);
196 }
197
198 int sum = std::accumulate(outcomesForZ.begin(), outcomesForZ.end(), 0);
199 qreal zAngle = M_PI*sum;
200 updatedAngle = xAngle + zAngle;
201
202 }
203
204 return updatedAngle;
