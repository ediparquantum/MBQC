53/*
54 inthe hand shake lemma stipulates that the total degree of a graph G,
55 is equivalent to 2 times the number of edges
56*/
57void assertHandShakeLemma(const undirectedGraph& graph) {
58
59 int num_vertices = boost::num_vertices(graph);
60 boost::container::vector<int> degree_vector;
61
62 // Call the function and print the adjacent vertices of vertex 0
63 for (int i = 0; i < num_vertices; i++) {
64 undirectedGraph::vertex_descriptor v;
65 v = boost::vertex(i, graph);
66 degree_vector.push_back(boost::degree(v, graph));
67 }
68
69 int num_edge = boost::num_edges(graph);
70 int sum = std::accumulate(degree_vector.begin(), degree_vector.end(), 0);
71 assert(("total degree is two times the edge set" && sum == 2 * num_edge));
72}
